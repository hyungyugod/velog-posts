public int[] solution(int[][] score) {

    int scoreln = score.length;

    List <Integer> box = new ArrayList<>();
    int[] answer = new int[scoreln];

    for (int i =0; i < scoreln; i++){

    for (int i =0; i < scoreln; i++) {
