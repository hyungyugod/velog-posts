# 📌 0. 리액트 라이브러리 활용
### 📌 0-1. sweetalert
- yarn add sweetalert2 sweetalert2-react-content 로 sweet alert 관련 페키지를 설치한다.
- onButtonClick1 함수에 Swal.fire를 정의하여 창을 열고 내부에 들어갈 내용을 정의한다.
- 이후  `onClick={onButtonClick1}`으로 연결하여 알림창을 띄운다.
```js
/**
 * SweetAlert2
 * 
 * https://sweetalert2.github.io/
 * 
 * yarn add sweetalert2 sweetalert2-react-content
 */
import React, { memo, useCallback } from "react";
import Swal from "sweetalert2";
import withReactContent from "sweetalert2-react-content";
import styled from "styled-components";

import img1 from "../../assets/img/img1.png";

const SweetAlertExContainer = styled.div`
    button {
        padding: 10px 20px;
        margin: 10px;
        font-size: 16px;
        font-weight: bold;
    }
`;

const SweetAlertEx = memo(() => {
    // Sweet Alert 모달창
    const MySwal = withReactContent(Swal);

    // 기본 알림창
    const onButtonClick1 = useCallback(() => {
        Swal.fire('안녕하세요');
    }, []);

    // 제목과 내용이 있는 알림창
    const onButtonClick2 = useCallback(() => {
        Swal.fire(
            '성공!',
            '작업이 성공적으로 완료되었습니다.',
            'success'
        );
    }, []);

    // 사용자 확인을 구하는 경고창
    const onButtonClick3 = useCallback(() => {
        Swal.fire({
            title: '정말로 삭제하시겠습니까?',
            text: "삭제된 데이터는 복구할 수 없습니다!",
            icon: 'warning',
            showCancelButton: true,
            confirmButtonColor: '#3085d6',
            cancelButtonColor: '#d33',
            confirmButtonText: '삭제',
            cancelButtonText: '취소'
        }).then((result) => {
            console.log(result);
            
            if (result.isConfirmed) {
                Swal.fire('삭제됨!', '데이터가 삭제되었습니다.', 'success');
            } else if (result.isDismissed && result.dismiss === 'backdrop') {
                Swal.fire('취소됨', '삭제가 취소되었습니다.');
            }
        });
    }, []);

    // Promise 방식을 사용한 처리방법
    const onButtonClick4 = useCallback(() => {
        const swalWithBootstrapButtons = Swal.mixin({
            customClass: {
                confirmButton: 'btn btn-success',
                cancelButton: 'btn btn-danger'
            },
            buttonsStyling: false
        });

        swalWithBootstrapButtons.fire({
            title: '정말로 삭제하시겠습니까?',
            text: "삭제된 데이터는 복구할 수 없습니다!",
            icon: 'warning',
            showCancelButton: true,
            confirmButtonText: '네, 삭제!',
            cancelButtonText: '아니오, 취소!',
            reverseButtons: true
        }).then((result) => {
            if (result.isConfirmed) {
                swalWithBootstrapButtons.fire(
                    '삭제됨!',
                    '데이터가 삭제되었습니다.',
                    'success'
                );
            } else if (result.dismiss === Swal.DismissReason.cancel) {
                swalWithBootstrapButtons.fire(
                    '취소됨',
                    '데이터가 안전합니다 :)',
                    'error'
                );
            }
        });
    }, []);

    // Async-Await 방식을 사용한 처리방법
    const onButtonClick5 = useCallback(async () => {
        const { value: email } = await Swal.fire({
            title: 'Input email address',
            input: 'email',
            inputLabel: 'Your email address',
            inputPlaceholder: 'Enter your email address'
        });

        if (email) {
            Swal.fire(`Entered email: ${email}`);
        }
    }, []);

    // 이미지를 넣는 경우
    const onButtonClick6 = useCallback(() => {
        Swal.fire({
            title: "Sweet!",
            text: "Modal with a custom image.",
            imageUrl: img1,
            imageWidth: 400,
            imageHeight: 200,
            imageAlt: "Custom image"
        });
    }, []);

    // 토스트 방식 출력
    const onButtonClick7 = useCallback(() => {
        const Toast = Swal.mixin({
            toast: true,
            position: 'top-end',
            showConfirmButton: false,
            timer: 3000,
            timerProgressBar: true,
            didOpen: (toast) => {
                toast.addEventListener('mouseenter', Swal.stopTimer);
                toast.addEventListener('mouseleave', Swal.resumeTimer);
            }
        });

        console.log(Toast);

        if (result.isConfirmed) {
            Swal.fire('삭제됨!', '데이터가 삭제되었습니다.', 'success');
        } else if (result.isDismissed) {
            Swal.fire('취소됨', '삭제가 취소되었습니다.');
        } else if (result.isDismissed && result.dismiss === 'backdrop') {
            Swal.fire('취소됨', '삭제가 취소되었습니다.');
        }
    }, []);

    // 출력내용
    const onButtonClick8 = useCallback(() => {
        MySwal.fire({
            icon: 'success',
            title: <strong>Success React 적용</strong>,
            html: <i>You can use <b>bold text</b>, <a href="//sweetalert2.github.io">links</a> and other HTML tags</i>,
            showCloseButton: true,
            showCancelButton: true,
            focusConfirm: false,
            confirmButtonText: '<i class="fa fa-thumbs-up"></i> Great!',
            confirmButtonAriaLabel: 'Thumbs up, great!',
            cancelButtonText: '<i class="fa fa-thumbs-down"></i>',
            cancelButtonAriaLabel: 'Thumbs down'
        });
    }, [MySwal]);

    return (
        <SweetAlertExContainer>
            <h1>SweetAlert2Ex</h1>

            <button className="btn" onClick={onButtonClick1}>Basic</button>
            <button className="btn" onClick={onButtonClick2}>Success Alert</button>
            <button className="btn" onClick={onButtonClick3}>Delete Confirm</button>
            <button className="btn" onClick={onButtonClick4}>Button Style</button>
            <button className="btn" onClick={onButtonClick5}>Input Dialog</button>
            <button className="btn" onClick={onButtonClick6}>Custom Image</button>
            <button className="btn" onClick={onButtonClick7}>Toast</button>
            <button className="btn" onClick={onButtonClick8}>React Content</button>
        </SweetAlertExContainer>
    );
});

export default SweetAlertEx;
```

### 📌 0-2. chartjs
- chartjs를 리액트 버전으로 다시 만든 라이브러리이다. 
- yarn add react-chartjs-2 chart.js
- Chart와 Chart의 구성 요소들을 모두 import한 다음에 Chart에서 register를 호출하여 Chart의 구성요소들을 Chart에 연결시켜야 한다.
- `import { Line, Bar, Pie, Radar } from 'react-chartjs-2';`를 이용하여 다양한 종류의 그래프들을 가져온다.
- legend는 legere + end로 읽을 수 읽는 것을 의미한다. (전설은 읽혀내려오는 이야기이다.)
- 지도, 도표 등에서의 legend는 이 기호가 의미하는 것을 읽어주는 해설이라는 뜻으로 범례라고도 쓰인다.
- ArcElement는 Arc가 호를 의미하기 때문에 도넛 차트를 그리는 도구를 의미한다.
- labels는 따로 설정하지 않는 한 가로축이고 datasets은 한계체가 가지는 값들의 배열이다.
- 사용할 곳에서 `<Line data={lineData} options={chartOptions} />` 처럼 필요한 컴포넌트를 꺼내서 사용하면 그래프를 구현할 수 있다.

    CategoryScale,
    LinearScale,
    PointElement,
    LineElement,
    Title,
    Tooltip,

    ArcElement,

    CategoryScale,

    Tooltip,

    BarElement,
    ArcElement,
);
